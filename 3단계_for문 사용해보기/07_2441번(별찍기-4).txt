#include<stdio.h>

int n;
int main()
{
	scanf("%d",&n);
	
	for(int i=1; i <= n; i++)  
	{
		for(int j=2;j<=i;j++)  
		{
			
			printf(" ");
			
		}
		for(int j=i; j<=n; j++)    
		{
			printf("*");
		}
		
		printf("\n");
		
	}

}

#include<stdio.h>

int n;
int main()
{
	scanf("%d",&n);
	
	for(int i=n; i >0; i--)  
	{
		for(int j=i+1; j<=n;j++)  
		{
			
			printf(" ");
			
		}
		for(int j=1; j<=i; j++)    
		{
			printf("*");
		}
		
		printf("\n");
		
	}

}
